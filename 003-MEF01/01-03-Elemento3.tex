    \begin{split}
      f_1^3 &= -k_3 (x_4 - x_3)\\[10pt]
      f_2^3 &= k_3 (x_4 - x_3)
    \end{split}
    \quad \Rightarrow \quad
     \begin{pmatrix}
       f_1^2\\[10pt]
       f_2^2
     \end{pmatrix}
     =
     \underbrace{
       k_3 
       \begin{pmatrix}
	 1 & -1 \\[10pt]
	 -1 & 1 
       \end{pmatrix}
     }_{ \mathbf{ k_3 ^{el} } }
    \begin{pmatrix}
      x_3 \\[10pt]
      x_4
    \end{pmatrix}
