\begin{equation}
  \begin{aligned}
    m R \dfrac{d^2\theta}{dt} + mg sin(\theta)=0 \\
    v = \dfrac{d\theta}{dt}\\
    \dfrac{dv}{dt} = -\dfrac{g}{R} sin(\theta) \\
    \dfrac{d}{dt} 
    \begin{bmatrix} \theta\\  v \end{bmatrix}
    =
    \begin{bmatrix} v \\ - \dfrac{g}{m} sin(\theta) \end{bmatrix}
  \end{aligned}
\end{equation}
